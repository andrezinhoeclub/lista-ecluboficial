// --- FUNÇÃO DE FINALIZAÇÃO DE RESERVA (COLE ESTE BLOCO) --- async function finalizarReserva() { // 1. Localiza os elementos da tela const btn = document.getElementById('btn-confirmar'); const nomeCliente = document.getElementById('input-nome').value.trim(); const telefoneCliente = document.getElementById('input-tel').value.trim(); const tipoIngresso = document.querySelector('input[name="tipo"]:checked')?.value; // CASAL ou SOLTEIRO const metodoPagamento = document.querySelector('input[name="pagamento"]:checked')?.value; // pix_antecipado ou portaria // 2. Validação básica if (!nomeCliente || !telefoneCliente || !tipoIngresso || !metodoPagamento) { alert("Ops! Preencha todos os campos e selecione a forma de pagamento."); return; } // Travamos o botão para o cliente não clicar duas vezes const originalText = btn.innerText; btn.innerText = "GERANDO PEDIDO..."; btn.disabled = true; try { // 3. Salva a reserva no seu Banco de Dados (Firestore) const docRef = await addDoc(collection(db, "lista_eventos"), { cliente: nomeCliente, telefone: telefoneCliente, evento: eventoAtivo.nome, data_evento: eventoAtivo.data, tipo: tipoIngresso, promoter: promoterNome.toUpperCase(), metodo: metodoPagamento, entrou: false, pago: false, data_reserva: new Date().toISOString() }); // 4. LÓGICA DE PAGAMENTO if (metodoPagamento === 'pix_antecipado') { // Pega o valor da Dashboard (ex: "R$ 50,00") e limpa para enviar (ex: 50.00) const valorRaw = tipoIngresso === 'CASAL' ? eventoAtivo.ant_casal : eventoAtivo.ant_solteiro; const valorLimpo = valorRaw.replace('R$', '').replace(/\s/g, '').replace(',', '.'); // URL da sua Cloud Function que você acabou de subir const URL_PONTE_MP = 'https://us-central1-teste-de-reserva.cloudfunctions.net/gerarPagamento'; const response = await fetch(URL_PONTE_MP, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ valor: valorLimpo, titulo: `E-CLUB: ${eventoAtivo.nome} (${tipoIngresso})`, reservaId: docRef.id // Envia o ID da reserva para o Mercado Pago }) }); const result = await response.json(); if (result.url) { // REDIRECIONAMENTO MÁGICO: Joga o cliente para o Checkout do Mercado Pago window.location.href = result.url; } else { throw new Error("Erro ao gerar link de pagamento."); } } else { // Se for portaria, apenas avisa e manda para a página de sucesso simples alert("Reserva confirmada! Seu nome está na lista da portaria."); window.location.href = "sucesso_lista.html"; } } catch (error) { console.error("Erro no processo:", error); alert("Ocorreu um erro técnico. Tente novamente ou chame o suporte no WhatsApp."); btn.innerText = originalText; btn.disabled = false; } }
